### Complie it
```
export LDFLAGS="-lfreetype -lformw -lfreetype -lm"
export CFLAGS="-I./include -I/usr/include/libpng16 -I/usr/include/freetype2"
gcc $CFLAGS -o bmh  src/ft_face.c src/main.c src/bmh/bmh_data.c $LDFLAGS
```
